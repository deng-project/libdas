62 /**
63 * Write a numerical value to the stream
64 * @param _value_name is a value name string that is used to define the value
65 * @param _value is a value in type T that is outputted to stream
66 */
67 template <typename T>
68 void _WriteNumericalValue(const std::string &_value_name, T _value) {
69 LIBDAS_ASSERT(std::is_floating_point<T>::value || std::is_integral<T>::value);
70 LIBDAS_ASSERT(m_out_stream.is_open());
71
72 m_out_stream.write(_value_name.c_str(), _value_name.size());
73 m_out_stream.write(": ", 2);
74 m_out_stream.write(reinterpret_cast<const char*>(&_value), sizeof(T));
75 m_out_stream.write(LIBDAS_DAS_NEWLINE, strlen(LIBDAS_DAS_NEWLINE));
76 }
77 /**
78 * Write an array value to the stream
79 * @param _value_name is a value name string that is used to define the value
80 * @param _n is the total value count
81 * @param _values is a T pointer value to the memory area that contains all the array values
82 */
83 template<typename T>
84 void _WriteArrayValue(const std::string &_value_name, uint32_t _n, T *_values) {
85 LIBDAS_ASSERT(m_out_stream.is_open());
86
87 m_out_stream.write(_value_name.c_str(), _value_name.size());
88 m_out_stream.write(": ", 2);
89 m_out_stream.write(reinterpret_cast<const char*>(_values), sizeof(T) * _n);
90 }
91 /**
92 * Write a generic data value to the stream
93 * @param _data is a pointer to the valid data that will be written
94 * @param _len is a length of the data in bytes
95 * @param _append_nl is a boolean flag indicating if the appendix newline should be written or not
96 * @param _value_name is a value name string that is used to define the value

98 void _WriteGenericDataValue(const char *_data, const size_t _len, bool _append_nl = true, const std::string &_value_name = "");
99 /**
100 * Write a generic 4x4 matrix value to the stream
101 * @param _value_name is a value name string that is used to define the value
102 * @param _mat is a reference to Matrix4<T> object that is written to the stream
103 */
104 template <typename T>
105 void _WriteMatrixValue(const std::string &_value_name, const Matrix4<T> &_mat) {
106 LIBDAS_ASSERT(std::is_floating_point<T>::value || std::is_integral<T>::value);
107 LIBDAS_ASSERT(m_out_stream.is_open());
108
109 m_out_stream.write(_value_name.c_str(), _value_name.size());
110 m_out_stream.write(": ", 2);
111
112 for(auto it = _mat.BeginRowMajor(); it != _mat.EndRowMajor(); it++) {
113 m_out_stream.write(reinterpret_cast<const char*>(it.GetData()), sizeof(T));
114 }
115
116 m_out_stream.write(LIBDAS_DAS_NEWLINE, strlen(LIBDAS_DAS_NEWLINE));
117 }
